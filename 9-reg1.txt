    <h2>Detailed Job Description / Requirements</h2>
    <div class="detail-text">
      <p><span style="font-size: 10pt;"><strong>Job Description:</strong></span></p>
<ul>
<li>Interface with Project Managers to understand product requirements and create a development plan.</li>
<li>Develop new features and fix bugs in existing projects.</li>
<li>Work with QA and build engineers to ensure the application can be built properly and tested.</li>
<li>Produce technical documentation for new features implemented.</li>
</ul>
<p><span style="font-size: 10pt;"><strong>Job Specification:</strong></span></p>
<ul>
<li>Experience on Java/J2EE.</li>
<li>Experience with Version control e.g. SVN.</li>
<li>Participate in product development using an agile, iterative process.</li>
<li>&nbsp;Familiar with J2EE, JPA,Hibernate, JDBC,JSF&nbsp;,Rich Faces,&nbsp;J Query,Web services.</li>
<li>Strong Object Oriented Programming skill, including MVC, Delegation, Singleton, Observer and Factory Design Pattern.</li>
<li>Experience with Memory Management, Thread, REST, JSON/XML Parsing.</li>
<li>Build in quality, reusable code and processes to all development activity.</li>
<li>Excellent verbal and written communication skills.</li>
<li>Strong analytical, debugging and troubleshooting skills and ability to work under a tight schedule and meet deadlines.</li>
</ul>
<p>&nbsp;</p>
<p><strong>F1soft International</strong> is looking for Junior Java Programmer&nbsp;for the head office located in Kathmandu.</p>
<p>Intetrested Candidates with Bachelor&rsquo;s degree in Computer Science, Information Technology/Management or related field. having one year or more experience can forward your CV at <strong>hr@f1soft.com.</strong></p>

optimize_theHotList();

function optimize_theHotList()
{
	centerContent_height = $('#main-content-wrapper').height();
	hotJobBox_height = $('#hot-jobs-content-holder').outerHeight();
	hjHeader_height = $('#hotjobs-box-header').height();
	
	if(hotJobBox_height > centerContent_height)
	{
		$('#hotjobList').height(centerContent_height - hjHeader_height-50);
	}
	$('.hotjobs-box').height($('#hotjobList').outerHeight() + hjHeader_height + 50);
}

function circulate_list(options) 
{
	if(!options.list.jquery)
		options.list = $(options.list);

	if(!options.listitem.jquery)
		options.listitem = $(options.listitem);

	if(!options.listitem.length) return;

	listitem_first = $(options.listitem[0]);	
	if( options.mini_list == true )
	{
		listitem_first.animate({
			'margin-top': '-' + listitem_first.outerHeight()
			}, speed, function(){
				options.list.append(listitem_first);
				listitem_first.css('margin-top', 0).addClass('on-the-way');
				//listitem_first.parents('.multiple').find('a.see-more').slideDown();
		});
	}
	else
	{
		if(listitem_first.hasClass('multiple') )
		{
			if(listitem_first.find('li:not(.on-the-way)').length > 1)
			{
				circulate_list({
					list: listitem_first.find('ul'),
					listitem: listitem_first.find('li'),
					speed: 'fast',
					mini_list:true
				});
			}
			else
			{
				listitem_first.find('a.see-more').slideUp();
				listitem_first.animate({
					'margin-top': '-' + listitem_first.outerHeight()
					}, speed, function(){
						options.list.append(listitem_first);
						listitem_first.css('margin-top', 0);
						listitem_first.find('.on-the-way').each(function(){
							$(this).removeClass('on-the-way');
						});
				});	
			}
		}
		else
		{
			listitem_first.animate({
				'margin-top': '-' + listitem_first.outerHeight()
				}, speed, function(){
					options.list.append(listitem_first);
					listitem_first.css('margin-top', 0);
			});
		}
		
	}
}
the_interval = 0;
$(function(){

	the_interval = setInterval(function(){
		circulate_list({
			list: '#hotjobList', 
			listitem: '.company-item', 
			speed: speed});
	}, pause_time);

	$('#hotjobList .company-item').hover(function(ev){
	    clearInterval(the_interval);
	}, function(ev){
	    the_interval = setInterval(function(){
			circulate_list({
				list: '#hotjobList', 
				listitem: '.company-item', 
				speed: speed});
		}, pause_time);
	});

	$('a.see-more').click(function(){
		list = $(this).parents('.multiple');
		list.find('.on-the-way').removeClass('on-the-way');
		$(this).slideUp();
	});
});
